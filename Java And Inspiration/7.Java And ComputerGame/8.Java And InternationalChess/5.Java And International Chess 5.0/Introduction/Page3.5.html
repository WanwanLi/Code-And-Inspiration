private void setPositionForX()
{
 foreach(chess[i][j]=='.')
 {
  chess[i][j]='X';nextScore=this.getMinScoreForO();
  if(nextScore>maxScore)
  {
   maxScore=nextScore;bestRow=i;bestColumn=j;
  }
  chess[i][j]='.';
 }
 chess[bestRow][bestColumn]='X';
}
 private int getMinScoreForO()
{
 foreach(chess[i][j]=='.')
 {
  chess[i][j]='O';nextScore=this.getMaxScoreForX();
  if(nextScore<minScore)minScore=nextScore;
  chess[i][j]='.';
 }
 return minScore;
}
